## Step 2 — Create email sending lib

Create `lib/email/send.ts`:

```typescript
import { Resend } from 'resend';

const resend = new Resend(process.env.RESEND_API_KEY);
const FROM   = process.env.EMAIL_FROM  ?? 'Jocril <encomendas@jocril.pt>';
const ADMIN  = process.env.ADMIN_EMAIL ?? 'geral@jocril.pt';

export interface OrderEmailData {
  orderNumber: string;
  customerName: string;
  customerEmail: string;
  total: number;
  items?: Array<{ productName: string; quantity: number; unitPrice: number }>;
}

export async function sendOrderConfirmation(data: OrderEmailData): Promise<void> {
  // fire-and-forget
  void resend.emails.send({
    from: FROM,
    to: data.customerEmail,
    subject: `Encomenda #${data.orderNumber} recebida — Jocril`,
    html: orderConfirmationHtml(data),
  }).catch((e) => console.error('sendOrderConfirmation failed:', e));
}

export async function sendAdminNotification(data: OrderEmailData): Promise<void> {
  void resend.emails.send({
    from: FROM,
    to: ADMIN,
    subject: `Nova encomenda #${data.orderNumber} — ${data.customerName}`,
    html: adminNotificationHtml(data),
  }).catch((e) => console.error('sendAdminNotification failed:', e));
}

export async function sendPaymentReceived(data: { orderNumber: string; customerEmail: string; total: number; customerName: string }): Promise<void> {
  void resend.emails.send({
    from: FROM,
    to: data.customerEmail,
    subject: `Pagamento confirmado — Encomenda #${data.orderNumber}`,
    html: paymentReceivedHtml(data),
  }).catch((e) => console.error('sendPaymentReceived failed:', e));
}
```

Write HTML template functions in the same file. Keep them simple — plain text layout with the Jocril brand color (`#2dd4cd`). No complex responsive design.

Each template must include:
- Order number (prominent)
- Customer name
- Total amount in EUR
- "Obrigado pela sua encomenda" message in Portuguese
- Contact email for support

**Important legal note:** These emails are confirmation-only, NOT invoices. Do NOT include an itemised invoice or NIF. Invoice generation is handled by certified billing software (PHC).
